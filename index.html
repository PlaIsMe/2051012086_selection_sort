public void SeclectionSort(int M[], int n)
{
 int K = 0;
 int PosMin;
 while(K < n-1)
 {
 int Min = M[K];
 PosMin = K;
 for(int Pos = K + 1; Pos < n; Pos++)
 {
  if(Min > M[Pos])
  {
   Min = M[Pos];
   PosMin = Pos;
  }
 }
 int temp = M[K];
 M[K] = M[PosMin];
 M[PosMin] = temp;
 K++;
 }
}
